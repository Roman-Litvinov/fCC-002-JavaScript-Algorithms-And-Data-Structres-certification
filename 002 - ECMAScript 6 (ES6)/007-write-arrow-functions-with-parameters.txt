// To pass arguments into an arrow function,
// put them inside the parentheses, like so:
const doubler = (item) => item*2;
var arg = 4;
console.log(arg + " doubled equals " + doubler(arg));

// When there's only a single argument for the function,
// we can omit the parentheses. The below function is exactly
// the same as the doubler() function above:
const doubler1 = item => item*2;
var arg = 4;
console.log(arg + " doubled still equals " + doubler1(arg));

// When there are more than 1 parameters, we list them
// separated by commas, like so:
const summer = (a,b) => a+b;
var arg1 = 2, arg2 = 3;
console.log(arg1 + "+" + arg2 + " equals " + summer(arg1,arg2));
