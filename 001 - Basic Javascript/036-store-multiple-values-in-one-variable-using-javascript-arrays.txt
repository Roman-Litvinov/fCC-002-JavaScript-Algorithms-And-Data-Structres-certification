// ARRAYS
// Array variables store several pieces of data in one place.
// Arrays are declared by enclosing array elements within
// square brackets and separating them by commas, like so:

var sandwichArray = ["peanut butter", "jelly", "bread"];
console.log(sandwichArray);

// Arrays can contain different data types, like strings and 
// numbers in the example below:
var myArray = ["this is a string", 5];

// When output to console, array elements are shown 
// enclosed in square brackets:
console.log(myArray);
